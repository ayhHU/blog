小程序静默登录注册

# 小程序静默登录、注册

# 背景

由于微信小程序 wx.getUserProfileapi不在返回用户头像名称信息、所以前端在登录的时候不再需要弹起授权框，在用户进去小程序的时候实现无感（静默）登录注册、基本uniapp框架、第三方库uview1.0。

# 封装源码

uniapp 获取小程序获取启动参数

/**获取启动小程序参数*/
class Params {
	
	/**获取海报携带的参数解析*/
	getParams(){
        const params = uni.getEnterOptionsSync()
        const scanParams = (params.query.params ? JSON.parse(decodeURIComponent(params.query.params)) : {})
        return  Object.assign(params.query,scanParams)
	}
}

export default Params;

静默登录的主要逻辑


/**账号状态(大于等10 都是被冻结状态(冻结时会再原本状态上加10))*/
const UserStatusEnums = {
	/**0-初始化状态;*/
	init:0,
	/**1待完善基础信息;*/
	await:1,
	/**2已完善账号正常账号;*/
	normal:2,
	/**3-申请删除中 */
	del:3
}

const appid = 'wx27822e8d41a38a5a'

/**登录方法*/
class Login {
	
	static vm = new Login()
	
	static getInstance(){return this.vm}
	/**暂时存token*/
	token = uni.getStorageSync('token') || ''
	/**
	 * 小程序登录
	 * @param askCode {String} 登录携带的用户邀请码
	 */
	onLogin(askCode = ''){
		return new Promise((resolve,reject)=>{
			/**如果本地有token就直接退出因为vuex写了提取本地的token和用户信息*/
			if(this.token){
				resolve()
				return;
			};
			uni.login({
				provider:'weixin',
				success:(resCode) => {
                      
                      //如果登录注册接口只需要一个code值、可以少了uni.getUserInfo这步
					uni.getUserInfo({
						success:async user=>{
							const {encryptedData,iv,} = user
							let data = {
								code : resCode.code,
								loginVersion:'1.0.1',
								encryptedData,
								iv
							}
							if(askCode) data['askCode'] = askCode
							/**调用登录注册接口*/
							let result = await uni.$u.api.onLogin(data)
							this.token = result.access_token
							/**获取用户信息*/
							const userInfo = await this.getUserInfo()
							resolve({...userInfo})
						}
					})
				}
			})
		})
	}
	
	/**获取用户信息*/
	async getUserInfo(){
		const res = await uni.$u.get('/user/app/user_info/my/info',{},{Authorization:`Bearer ${this.token}`})
		//判断是否已经完成登录了
        //如果没有完成用户信息的完善、就是没登录、实质上已经注册了
		if(res.user.accountStatus != UserStatusEnums.normal) return {
			isLogin:false,
			data:res.user,
			token:this.token
		}
        //如果已经完成注册和完善信息那么就是登录成功了
        //保存用户信息到本地
		uni.setStorageSync('token',this.token)
		uni.setStorageSync('info',res.user)
		return {isLogin:true,data:res,token:this.token}
	}
	
	/**编辑用户信息*/
	async editUserInfo(info){
		await uni.$u.put('/user/app/user_info/my/edit',info,{Authorization:`Bearer ${this.token}`})
		uni.showToast({
			icon:'none',
			title:'编辑成功'
		})
	}
}

export default Login

# 使用

在app.vue里onShow生命周期里添加

//获取启动参数
const params = new Params()
const scanParams = params.getParams()
let {askCode} = scanParams
/**静默登录*/
const login = Login.getInstance()
login.onLogin(askCode).then(res=>{
    //如果登录成功的操作
    if(res && res.isLogin){
        this.user.token = res.token
        this.user.info = res.data.user
        this.user.isLogin = true
    }
    //如果并且没有绑定分销的、并且有邀请码的
    if(this.user.token && askCode){
        //绑定分销
        api.onBindDistribution({
            askCode:askCode
        })
    }
})
